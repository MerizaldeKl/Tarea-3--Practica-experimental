function Contar() {
let numeros = [6, 10, 28];
let perfectos = [];

for (let i = 0; i < numeros.length; i++) {
    let num = numeros[i];
    let suma = 0;

    
    for (let n = 1; n < num; n++) {
        if (num % n === 0) {
            suma += n;
        }
    }

    if (suma === num) {
        perfectos.push(num);
    }
}

alert(`Números perfectos: ${perfectos.join(", ")}`);
}
Contar()
