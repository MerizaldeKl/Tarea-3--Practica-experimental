function Contar() {
    let palabra = prompt("Ingrese la palabra");
    let cantidad = 0;
    if(palabra) {
        cantidad = palabra.length;
    alert(`La palabra tiene ${cantidad} caracteres`);
    } else {
        alert(`No ingreso palabras`)
    }
}
