function Contar() {
let oracion = prompt("Ingrese una oración:");
let palabra = "";

for (let i = 0; i < oracion.length; i++) {
    let caracter = oracion[i];
    
    if (caracter !== " ") {
        palabra += caracter; 
    } else {
        alert(palabra); 
        palabra = ""; 
    }
}

if (palabra !== "") {
    alert(palabra);
}
}
Contar()
