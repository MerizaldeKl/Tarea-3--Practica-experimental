function Contar() {
let numeros = [3, 5];

for (let i = 0; i < numeros.length; i++) {
    let num = numeros[i];
    let factorial = 1;

    for (let n = 1; n <= num; n++) {
        factorial *= n; 
        
    }

    alert(`${num}! = ${factorial}`);
}
}
Contar()
