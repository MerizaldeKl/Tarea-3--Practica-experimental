function agregarAlInicio(texto, fragmento) {
    let resultado = "";

    for (let i = 0; i < fragmento.length; i++) {
        resultado = resultado + fragmento[i];
    }

    resultado = resultado + " ";

    for (let i = 0; i < texto.length; i++) {
        resultado = resultado + texto[i];
    }

    return resultado;
}

function agregarAlFinal(texto, fragmento) {
    let resultado = "";

    for (let i = 0; i < texto.length; i++) {
        resultado = resultado + texto[i];
    }

    resultado = resultado + " ";

    for (let i = 0; i < fragmento.length; i++) {
        resultado = resultado + fragmento[i];
    }

    return resultado;
}

function agregarTexto(texto, fragmento) {
    let alInicio = "";
    let alFinal = "";


    for (let i = 0; i < fragmento.length; i++) {
        alInicio = alInicio + fragmento[i];
    }
    alInicio = alInicio + " ";
    for (let i = 0; i < texto.length; i++) {
        alInicio = alInicio + texto[i];
    }

    for (let i = 0; i < texto.length; i++) {
        alFinal = alFinal + texto[i];
    }
    alFinal = alFinal + " ";
    for (let i = 0; i < fragmento.length; i++) {
        alFinal = alFinal + fragmento[i];
    }

    return {
        inicio: alInicio,
        final: alFinal
    };
}
