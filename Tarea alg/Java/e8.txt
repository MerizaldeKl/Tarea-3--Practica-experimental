function Contar() {
let frase1 = prompt("Ingrese la primera frase:");
let frase2 = prompt("Ingrese la segunda frase:");
if(!frase1 || !frase2) {
    alert(`No ingresaste la frase 1 o 2.`);
    return;
}

let contador1 = 0;
let contador2 = 0;

for (let i = 0; i < frase1.length; i++) {
    if (frase1[i] != " ") {
        contador1++;
    }
}

for (let i = 0; i < frase2.length; i++) {
    if (frase2[i] != " ") {
        contador2++;
    }
}

if (contador1 > contador2) {
    alert(`La frase '${frase1}' tiene más caracteres.`);
} else if (contador2 > contador1) {
    alert(`La frase '${frase2}' tiene más caracteres.`);
} else {
    alert("Ambas frases tienen la misma cantidad de caracteres.");
}
}
Contar();
