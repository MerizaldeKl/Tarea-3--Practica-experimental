function contarParrafos(texto) {
    let parrafos = 1;

    for (let i = 0; i < texto.length; i++) {
        if (texto[i] === '\n') {
            parrafos++;
        }
    }

    return parrafos;
}

let texto = "Hola mundo\nEsto es un segundo párrafo";
console.log("Número de párrafos: " + contarParrafos(texto));
