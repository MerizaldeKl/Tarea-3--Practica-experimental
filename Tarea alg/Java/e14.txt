function Contar() {
let numeros = [7, 8, 13];
let primos = []; 

for (let i = 0; i < numeros.length; i++) {
    let num = numeros[i];
    let esPrimo = true;

    if (num < 2) {
        esPrimo = false; 
    } else {
        
        for (let n = 2; n < num; n++) {
            if (num % n === 0) {
                esPrimo = false;
                break;
            }
        }
    }

    if (esPrimo) {
        primos.push(num);
    }
}

alert(`Números primos: ${primos.join(", ")}`);
}
Contar()
