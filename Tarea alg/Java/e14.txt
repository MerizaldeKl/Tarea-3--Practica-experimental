function contarPosicionesImpares(texto) {
    let contador = 0;
    let caracteresImpares = [];

    for (let i = 0; i < texto.length; i++) {
        if (i % 2 !== 0) {
            contador++;
            caracteresImpares.push(texto[i]);
        }
    }

    return contador;
}

console.log(contarPosicionesImpares("Hola")); // 2
