function contarPalabrasMayuscula(texto) {
    let contador = 0;
    let dentroDePalabra = false;

    for (let i = 0; i < texto.length; i++) {
        let c = texto[i];

        if (c !== " " && c !== "\n") {
            if (dentroDePalabra === false) {
                if (c >= "A" && c <= "Z") {
                    contador++;
                }

                dentroDePalabra = true; 
            }
        } else {
            dentroDePalabra = false;
        }
    }

    return contador;
}
