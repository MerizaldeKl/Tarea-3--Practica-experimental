function Contar() {
  let texto = prompt("Ingrese un texto:");
  let letra = prompt("Ingrese una letra");
  if (!texto || !letra) {
    alert("No ingresaste un texto o letra.");
    return;
  }

  let textoModificado = "";

  for (let i = 0; i < texto.length; i++) {
    if (texto[i].toLowerCase() === letra.toLowerCase()) {
      textoModificado += "#";
    } else {
      textoModificado += texto[i];
    }
  }

  alert(`El texto modificado es: ${textoModificado}`);
}
Contar()
