function Contar() {
  let texto = prompt("Ingrese una frase:");
  let letra = prompt("Ingrese la letra que desea contar:");

  let contador = 0;

  for (let i = 0; i < texto.length; i++) {
    if(texto[i].toLowerCase() === letra.toLowerCase()) {
      contador++;
    } 
  }
   alert(`La letra ${letra} aparece ${contador} veces.`);
}
Contar()
