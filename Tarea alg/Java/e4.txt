function Contar() {
  let apellido1 = prompt("Ingrese el primer apellido:");
  let apellido2 = prompt("Ingrese el segundo apellido:");

  if (!apellido1 || !apellido2) {
    alert("No ingresaste uno o ambos apellidos.");
    return;
  }

  if (apellido1.length > apellido2.length) {
    alert(`El apellido '${apellido1}' tiene más letras.`);
  } else if (apellido2.length > apellido1.length) {
    alert(`El apellido '${apellido2}' tiene más letras.`);
  } else {
    alert(`Los apellidos '${apellido1}' y '${apellido2}' tienen la misma cantidad de letras.`);
  }
}
Contar()
