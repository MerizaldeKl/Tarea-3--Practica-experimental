function buscarPalabra(texto, palabra) {
    let match = 0; 

    for (let i = 0; i < texto.length; i++) {

        if (texto[i] === palabra[match]) {
            match++; 

            if (match === palabra.length) {
                return true;
            }
        } else {
            match = 0;
        }
    }

    return false;
}

let texto = "Me gusta programar en JavaScript";
let palabra = "programar";

if (buscarPalabra(texto, palabra)) {
    console.log("La palabra '" + palabra + "' sÃ­ se encuentra en el texto.");
} else {
    console.log("La palabra '" + palabra + "' NO se encuentra.");
}
