function Contar() {
  let nombre = prompt("Ingrese su nombre completo:");
  if (!nombre) {
    alert("No ingresaste un nombre.");
    return;
  }

  let iniciales = "";  
  
  for (let i = 0; i < nombre.length; i++) {
  
    if (i === 0 || nombre[i - 1] === " ") {
      iniciales += nombre[i].toUpperCase() + ".";
    }
  }

  alert(`Las iniciales son: ${iniciales}`);
}
Contar()
