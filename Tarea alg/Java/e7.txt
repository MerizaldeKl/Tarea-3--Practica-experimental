function contarPalabrasMinuscula(texto) {
    let contador = 0;
    let nuevaPalabra = true;

    for (let i = 0; i < texto.length; i++) {
        let c = texto[i];

        if (c === " " || c === "\n") {
            nuevaPalabra = true; 
        } else {
            if (nuevaPalabra === true) {

                if (c >= 'a' && c <= 'z') {
                    contador++;
                }

                nuevaPalabra = false; // ya no es inicio
            }
        }
    }

    return contador;
}
