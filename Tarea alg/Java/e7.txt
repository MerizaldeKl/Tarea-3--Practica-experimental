let texto = "hola Mundo bonito";
let contador = 0;
let enPalabra = false;

for (let i = 0; i < texto.length; i++) {
    let c = texto[i];

    if (c !== " " && enPalabra === false) {
        enPalabra = true;

        if (c >= 'a' && c <= 'z') {
            contador++;
        }
    }

    if (c === " ") {
        enPalabra = false;
    }
}

console.log("Palabras con minÃºscula inicial: " + contador);
