function Contar() {
let texto = prompt("Ingrese un texto:");
let caracter = prompt("Ingrese el carácter que desea contar:");
if(!texto || !caracter) {
    alert(`No ingresaste texto o caracter`);
    return;
}

let contador = 0;

for (let i = 0; i < texto.length; i++) {
    if (texto[i].toLowerCase() === caracter.toLowerCase()) {
        contador++;
    }
}

alert(`La letra '${caracter}' aparece ${contador} veces.`);
}
Contar()
