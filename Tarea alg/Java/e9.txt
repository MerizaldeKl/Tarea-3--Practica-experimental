function invertirTexto(texto) {
    let resultado = [];

    for (let i = texto.length - 1; i >= 0; i--) {
        resultado.push(texto[i]);  
    }

    let invertido = "";

    for (let j = 0; j < resultado.length; j++) {
        invertido = invertido + resultado[j];
    }

    return invertido;
}

let texto = "Hola";
console.log("Invertido: " + invertirTexto(texto));
